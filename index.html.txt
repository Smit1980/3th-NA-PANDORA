      var usedQuestions = [];
      var buttonNames = [
        "С Благодарностью",
        "От Смита",
        "С Любовью"
      ];
      var currentButtonNameIndex = 0;
      
      function getRandomQuestion() {
        if (usedQuestions.length === questions.length) {
          alert("Вопросы закончились. Нажмите Сбросить, чтобы начать заново.");
          return;
        }
        var index;
        do {
          index = Math.floor(Math.random() * questions.length);
        } while (usedQuestions.includes(index));
        usedQuestions.push(index);
        document.getElementById("question").textContent = questions[index];
      }
      
      function resetQuestions() {
        usedQuestions = [];
        document.getElementById("question").textContent = "";
      }
      
      function changeButtonName() {
        currentButtonNameIndex = (currentButtonNameIndex + 1) % buttonNames.length;
        document.getElementsByTagName("button")[2].textContent = buttonNames[currentButtonNameIndex];
      }
